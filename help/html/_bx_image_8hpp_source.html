00998                 4, 4, 4, 4, 5, 5, 5, 5, 0};
00999             global_data const int DistanceBase[30] = {
01000                    1,     2,     3,     4,     5,     7,     9,    13,    17,    25,
01001                   33,    49,    65,    97,   129,   193,   257,   385,   513,   769,
01002                 1025,  1537,  2049,  3073,  4097,  6145,  8193, 12289, 16385, 24577};
01003             global_data const int DistanceExtra[30] = {
01004                 0,  0,  0,  0,  1,  1,  2,  2,  3,  3,
01005                 4,  4,  5,  5,  6,  6,  7,  7,  8,  8,
01006                 9,  9, 10, 10, 11, 11, 12, 12, 13, 13};
01007 
01008             // LZ윈도우의 복사거리동안 디코딩은 쉼
01009             if(0 < CopyCount)
01010             {
01011                 byte Value8 = LZWindow[CopyPos];
01012                 LZWindow[LZWindowPos] = Value8;
01013                 --CopyCount;
01014                 CopyPos = (CopyPos + 1) & 0x7fff;
01015                 LZWindowPos = (LZWindowPos + 1) & 0x7fff;
01016                 return Value8;
01017             }
01018 
01019             // 허프만트리에 입각하여 하나의 값을 읽어냄
01020             uint Value32 = Decode(0);
01021 
01022             if(Value32 == 256)
01023             {
01024                 if(!IsFinalBlock)
01025                     DecodeBlockHeader();
01026                 return DecodeByte();
01027             }
01028             else if(Value32 < 256) // LZ77상의 기호토큰인 경우

01036             uint Length = LengthBase[Value32 - 257] + GetBits(LengthExtra[Value32 - 257]);
01037             Value32 = Decode(1);
01038             uint Distance = DistanceBase[Value32] + GetBits(DistanceExtra[Value32]);
01039 
01040             CopyPos = (32768 + LZWindowPos - Distance) & 0x7fff;
01041             CopyCount = Length;
01042 
01043             return DecodeCompressedByte();
